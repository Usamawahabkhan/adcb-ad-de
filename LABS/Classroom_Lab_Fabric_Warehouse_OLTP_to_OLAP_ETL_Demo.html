Learning Objectives

  • Understand how detailed source data can be transformed into analytical summary data.
  • Create staging, clean, and fact-style summary tables in Microsoft Fabric Warehouse.
  • Run data quality checks before loading curated output.
  • Use aggregation to create an OLAP-style summary table.
  • Use MERGE as an incremental loading pattern.
  • Run audit checks to compare raw, staged, clean, and summary outputs.

Concepts Covered

  • OLTP-like source: Detailed transactional source table such as dbo.Trip.
  • Staging: Source-shaped table used before data cleaning.
  • Validation: SQL checks for duplicate, missing, or invalid data.
  • Cleaning: Standardizing values and filtering invalid records.
  • OLAP summary: Aggregated table optimized for analytics and reporting.
  • MERGE: Incremental load pattern for updating or inserting summary rows.
  • Audit: Row-count and revenue checks after loading.

End-to-End Flow

dbo.Trip sample table ↓ stage_trip_delta ↓ validation checks ↓ stage_trip_clean ↓ fact_trip_summary ↓ join to dbo.Geography ↓ business revenue query ↓ MERGE incremental load ↓ audit checks

Before You Run

  • Open the Microsoft Fabric Warehouse sample in the SQL Query Editor.
  • Run each step in order.
  • If your sample table has different column names, first run SELECT TOP 10 * FROM dbo.Trip; and adjust the column names accordingly.
  • This lab creates only lab-owned tables and does not update the sample source tables.
Important: This lab is for Microsoft Fabric Warehouse, not local SQL Server. Use the Warehouse SQL Query Editor.

-- ============================================================
-- STEP 3: LOAD STAGING FROM SAMPLE WAREHOUSE TABLE

-- Purpose:
--   Load a manageable subset from dbo.Trip into the staging table.
--   TOP is used to keep classroom execution fast.

INSERT INTO dbo.stage_trip_delta (

    DateID,
    PickupTimeID,
    PickupGeographyID,
    PassengerCount,
    TotalAmount,

    TripDistanceMiles

    CAST(MedallionID AS VARCHAR(100)) AS MedallionID,

SELECT COUNT(*) AS StageRows
FROM dbo.stage_trip_delta;

STEP 4: VALIDATION / DATA QUALITY CHECKS

-- ============================================================
-- STEP 4: VALIDATION / DATA QUALITY CHECKS
-- ============================================================
-- Purpose:
--   Identify duplicates, missing values, and invalid business values before loading curated output.

-- 4A: Possible duplicate operational grain check
SELECT
    MedallionID,
    DateID,
    PickupTimeID,
    COUNT(*) AS DuplicateCount
FROM dbo.stage_trip_delta
GROUP BY MedallionID, DateID, PickupTimeID
HAVING COUNT(*) > 1
ORDER BY DuplicateCount DESC;

-- 4B: Missing or invalid value checks
SELECT COUNT(*) AS MissingTotalAmountRows
FROM dbo.stage_trip_delta
WHERE TotalAmount IS NULL;

SELECT COUNT(*) AS InvalidDistanceRows
FROM dbo.stage_trip_delta
WHERE TripDistanceMiles <= 0 OR TripDistanceMiles IS NULL;

SELECT COUNT(*) AS MissingGeographyRows
FROM dbo.stage_trip_delta
WHERE PickupGeographyID IS NULL;

-- 4C: Row-level validation status sample
SELECT TOP 100
    MedallionID,
    DateID,
    PickupTimeID,
    PickupGeographyID,
    PassengerCount,
    TotalAmount,
    TripDistanceMiles,
    CASE
        WHEN DateID IS NULL THEN 'Invalid DateID'
        WHEN PickupGeographyID IS NULL THEN 'Invalid GeographyID'
        WHEN TotalAmount IS NULL THEN 'Missing TotalAmount'
        WHEN TripDistanceMiles <= 0 OR TripDistanceMiles IS NULL THEN 'Invalid TripDistanceMiles'
        ELSE 'Valid'
    END AS ValidationStatus
FROM dbo.stage_trip_delta
ORDER BY DateID, PickupTimeID;

STEP 5: CLEANING / CURATED STAGE

-- ============================================================
-- STEP 5: CLEANING / CURATED STAGE
-- ============================================================
-- Purpose:
--   Create a clean staging table with only valid rows and standardized values.

CREATE TABLE dbo.stage_trip_clean (
    DateID             INT           NOT NULL,
    PickupGeographyID  INT           NOT NULL,
    PassengerCount     INT           NULL,
    TotalAmount        DECIMAL(18,2) NOT NULL,
    PaymentType        VARCHAR(50)   NOT NULL,
    TripDistanceMiles  DECIMAL(18,2) NOT NULL
);

INSERT INTO dbo.stage_trip_clean (
    DateID,
    PickupGeographyID,
    PassengerCount,
    TotalAmount,
    PaymentType,
    TripDistanceMiles
)
SELECT
    DateID,
    PickupGeographyID,
    PassengerCount,
    TotalAmount,
    CASE
        WHEN PaymentType IS NULL OR LTRIM(RTRIM(PaymentType)) = '' THEN 'Unknown'
        ELSE PaymentType
    END AS PaymentType,
    TripDistanceMiles
FROM dbo.stage_trip_delta
WHERE DateID IS NOT NULL
  AND PickupGeographyID IS NOT NULL
  AND TotalAmount IS NOT NULL
  AND TripDistanceMiles > 0;

SELECT COUNT(*) AS CleanRows
FROM dbo.stage_trip_clean;

STEP 6: CREATE OLAP FACT SUMMARY TABLE

-- ============================================================
-- STEP 6: CREATE OLAP FACT SUMMARY TABLE
-- ============================================================
-- Purpose:
--   Create a summarized fact table for analytics.
--   Grain: one row per DateID and GeographyID.

CREATE TABLE dbo.fact_trip_summary (
    DateID        INT           NOT NULL,
    GeographyID   INT           NOT NULL,
    TotalTrips    INT           NOT NULL,
    TotalRevenue  DECIMAL(18,2) NOT NULL,
    AvgDistance   DECIMAL(18,2) NOT NULL
);

STEP 7: LOAD OLAP FACT SUMMARY TABLE

-- ============================================================
-- STEP 7: LOAD OLAP FACT SUMMARY TABLE
-- ============================================================
-- Purpose:
--   Aggregate clean trip records into an analytics-ready summary.

INSERT INTO dbo.fact_trip_summary (
    DateID,
    GeographyID,
    TotalTrips,
    TotalRevenue,
    AvgDistance
)
SELECT
    DateID,
    PickupGeographyID AS GeographyID,
    COUNT(*) AS TotalTrips,
    SUM(TotalAmount) AS TotalRevenue,
    AVG(TripDistanceMiles) AS AvgDistance
FROM dbo.stage_trip_clean
GROUP BY DateID, PickupGeographyID;

STEP 8: OLAP RESULT PREVIEW

-- ============================================================
-- STEP 8: OLAP RESULT PREVIEW
-- ============================================================
-- Purpose:
--   Confirm the summary table loaded successfully.

SELECT TOP 10
    DateID,
    GeographyID,
    TotalTrips,
    TotalRevenue,
    AvgDistance
FROM dbo.fact_trip_summary
ORDER BY TotalRevenue DESC;

STEP 9: BUSINESS QUERY WITH DIMENSION CONTEXT

-- ============================================================
-- STEP 9: BUSINESS QUERY WITH DIMENSION CONTEXT
-- ============================================================
-- Purpose:
--   Join the OLAP summary to Geography to answer a business question.
--   Business question: Which cities generated the highest revenue?

SELECT TOP 20
    g.City,
    SUM(f.TotalTrips) AS TotalTrips,
    SUM(f.TotalRevenue) AS Revenue,
    AVG(f.AvgDistance) AS AvgDistance
FROM dbo.fact_trip_summary AS f
JOIN dbo.Geography AS g
    ON f.GeographyID = g.GeographyID
GROUP BY g.City
ORDER BY Revenue DESC;

STEP 10: MERGE / INCREMENTAL LOAD PATTERN

-- ============================================================
-- STEP 10: MERGE / INCREMENTAL LOAD PATTERN
-- ============================================================
-- Purpose:
--   Demonstrate how a repeated load can update existing summary rows
--   and insert new rows when the source changes.

MERGE INTO dbo.fact_trip_summary AS tgt
USING (
    SELECT
        DateID,
        PickupGeographyID AS GeographyID,
        COUNT(*) AS TotalTrips,
        SUM(TotalAmount) AS TotalRevenue,
        AVG(TripDistanceMiles) AS AvgDistance
    FROM dbo.stage_trip_clean
    GROUP BY DateID, PickupGeographyID
) AS src
ON tgt.DateID = src.DateID
AND tgt.GeographyID = src.GeographyID
WHEN MATCHED THEN
    UPDATE SET
        tgt.TotalTrips = src.TotalTrips,
        tgt.TotalRevenue = src.TotalRevenue,
        tgt.AvgDistance = src.AvgDistance
WHEN NOT MATCHED THEN
    INSERT (DateID, GeographyID, TotalTrips, TotalRevenue, AvgDistance)
    VALUES (src.DateID, src.GeographyID, src.TotalTrips, src.TotalRevenue, src.AvgDistance);

STEP 11: AUDIT CHECKS

-- ============================================================
-- STEP 11: AUDIT CHECKS
-- ============================================================
-- Purpose:
--   Compare stage, clean, and fact outputs for simple operational validation.

SELECT
    COUNT(*) AS FactRows,
    SUM(TotalTrips) AS TotalTripsRepresented,
    SUM(TotalRevenue) AS TotalRevenue
FROM dbo.fact_trip_summary;

SELECT COUNT(*) AS RejectedRows_InvalidDistance
FROM dbo.stage_trip_delta
WHERE TripDistanceMiles <= 0 OR TripDistanceMiles IS NULL;

SELECT
    (SELECT COUNT(*) FROM dbo.Trip) AS RawTripTableRows,
    (SELECT COUNT(*) FROM dbo.stage_trip_delta) AS StageRowsLoaded,
    (SELECT COUNT(*) FROM dbo.stage_trip_clean) AS CleanRowsLoaded,
    (SELECT COUNT(*) FROM dbo.fact_trip_summary) AS OLAggregateRows;
