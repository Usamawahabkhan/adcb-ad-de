2. Learning Objectives

  • Understand how SQL is used in warehouse loading workflows.
  • Create simple stage and target tables.

3. Small Definitions

TermDefinition
ELTExtract, Load, Transform. Data is loaded first, then transformed in the warehouse.
Stage tableStores incoming/raw data before validation and curated loading.
Validation queryChecks data quality rules before loading curated output.
Target tableStores clean, validated, analytics-ready rows.
MERGESQL upsert pattern: update matching rows and insert new rows.
Materialized view patternPhysically stores repeated query output. In this lab: aggregate table + standard view.

STEP 2: CREATE STAGE TABLE

Creates the stage table that receives incoming/raw sales rows before validation.

-- ============================================================
-- STEP 2: CREATE STAGE TABLE
-- ============================================================
-- Stage table definition:
--   A stage table temporarily stores incoming/raw rows before they are
--   validated and loaded into curated target tables.
--
-- In real projects, stage tables are often loaded from files, pipelines,
-- dataflows, external systems, or lakehouse shortcuts.

CREATE TABLE dbo.stage_RetailSales (
    BatchID        INT           NOT NULL,
    SourceSystem   VARCHAR(50)   NOT NULL,
    OrderID        INT           NOT NULL,
    OrderLineID    INT           NOT NULL,
    OrderDate      DATE          NOT NULL,
    CustomerID     INT           NOT NULL,
    CustomerName   VARCHAR(200)  NOT NULL,
    ProductID      INT           NOT NULL,
    ProductName    VARCHAR(200)  NOT NULL,
    CategoryName   VARCHAR(100)  NOT NULL,
    StoreID        INT           NOT NULL,
    StoreName      VARCHAR(200)  NOT NULL,
    Quantity       INT           NOT NULL,
    UnitPrice      DECIMAL(10,2) NOT NULL,
    DiscountPct    DECIMAL(5,2)  NOT NULL,
    LoadDate       DATE          NOT NULL
);

STEP 4: VALIDATE STAGED DATA BEFORE LOADING

Runs validation queries to inspect row counts, duplicate business keys, and data quality status.

-- ============================================================
-- STEP 4: VALIDATE STAGED DATA BEFORE LOADING
-- ============================================================
-- Validation means checking whether staged data is complete, reasonable,
-- and safe to load into the curated target table.

-- 4A: Row count by batch and source system
SELECT
    BatchID,
    SourceSystem,
    COUNT(*) AS StageRowCount
FROM dbo.stage_RetailSales
GROUP BY BatchID, SourceSystem
ORDER BY BatchID, SourceSystem;

-- 4B: Duplicate business key check
-- Business key for this lab: OrderID + OrderLineID
SELECT
    OrderID,
    OrderLineID,
    COUNT(*) AS DuplicateCount
FROM dbo.stage_RetailSales
GROUP BY OrderID, OrderLineID
HAVING COUNT(*) > 1;

-- 4C: Data quality validation report
SELECT
    BatchID,
    OrderID,
    OrderLineID,
    CASE
        WHEN Quantity <= 0 THEN 'Invalid Quantity'
        WHEN UnitPrice <= 0 THEN 'Invalid Unit Price'
        WHEN DiscountPct < 0 OR DiscountPct > 100 THEN 'Invalid Discount Percent'
        WHEN CustomerID <= 0 THEN 'Invalid CustomerID'
        WHEN ProductID <= 0 THEN 'Invalid ProductID'
        WHEN StoreID <= 0 THEN 'Invalid StoreID'
        ELSE 'Valid'
    END AS ValidationStatus
FROM dbo.stage_RetailSales
ORDER BY OrderID, OrderLineID;

STEP 5: CAPTURE INVALID ROWS IN AN ERROR TABLE

Creates and loads an error table so invalid rows are captured with clear rejection reasons.

-- ============================================================
-- STEP 5: CAPTURE INVALID ROWS IN AN ERROR TABLE
-- ============================================================
-- Error table definition:
--   Stores rejected rows and a reason so the class can inspect data quality
--   problems without silently losing the bad data.

CREATE TABLE dbo.elt_LoadErrors (
    BatchID          INT           NOT NULL,
    SourceSystem     VARCHAR(50)   NOT NULL,
    OrderID          INT           NOT NULL,
    OrderLineID      INT           NOT NULL,
    ErrorReason      VARCHAR(200)  NOT NULL,
    ErrorCapturedDate DATE         NOT NULL
);

INSERT INTO dbo.elt_LoadErrors
SELECT
    BatchID,
    SourceSystem,
    OrderID,
    OrderLineID,
    CASE
        WHEN Quantity <= 0 THEN 'Invalid Quantity'
        WHEN UnitPrice <= 0 THEN 'Invalid Unit Price'
        WHEN DiscountPct < 0 OR DiscountPct > 100 THEN 'Invalid Discount Percent'
        WHEN CustomerID <= 0 THEN 'Invalid CustomerID'
        WHEN ProductID <= 0 THEN 'Invalid ProductID'
        WHEN StoreID <= 0 THEN 'Invalid StoreID'
        ELSE 'Unknown Error'
    END AS ErrorReason,
    CAST('2024-01-20' AS DATE) AS ErrorCapturedDate
FROM dbo.stage_RetailSales
WHERE Quantity <= 0
   OR UnitPrice <= 0
   OR DiscountPct < 0
   OR DiscountPct > 100
   OR CustomerID <= 0
   OR ProductID <= 0
   OR StoreID <= 0;

SELECT *
FROM dbo.elt_LoadErrors
ORDER BY OrderID, OrderLineID;

-- ============================================================
-- STEP 6: CREATE CURATED TARGET TABLE

STEP 7: MERGE VALID STAGE ROWS INTO TARGET

Uses MERGE to upsert valid staged rows into the curated target table using OrderID + OrderLineID as the match key.

-- ============================================================
-- STEP 7: MERGE VALID STAGE ROWS INTO TARGET
-- ============================================================
-- MERGE definition:
--   MERGE performs an upsert: update rows that already exist in the target
--   and insert rows that do not exist yet.
--
-- Matching rule:
--   OrderID + OrderLineID identify one sales order line.

MERGE INTO dbo.tgt_RetailSales AS target
USING (
    SELECT
        BatchID,
        SourceSystem,
        OrderID,
        OrderLineID,
        OrderDate,
        CustomerID,
        CustomerName,
        ProductID,
        ProductName,
        CategoryName,
        StoreID,
        StoreName,
        Quantity,
        UnitPrice,
        DiscountPct,
        ROUND(Quantity * UnitPrice, 2) AS GrossSalesAmount,
        ROUND(Quantity * UnitPrice * (DiscountPct / 100), 2) AS DiscountAmount,
        ROUND(Quantity * UnitPrice * (1 - DiscountPct / 100), 2) AS NetSalesAmount,
        LoadDate
    FROM dbo.stage_RetailSales
    WHERE Quantity > 0
      AND UnitPrice > 0
      AND DiscountPct >= 0
      AND DiscountPct <= 100
      AND CustomerID > 0
      AND ProductID > 0
      AND StoreID > 0
) AS source
ON target.OrderID = source.OrderID
AND target.OrderLineID = source.OrderLineID
WHEN MATCHED THEN
    UPDATE SET
        target.OrderDate = source.OrderDate,
        target.CustomerID = source.CustomerID,
        target.CustomerName = source.CustomerName,
        target.ProductID = source.ProductID,
        target.ProductName = source.ProductName,
        target.CategoryName = source.CategoryName,
        target.StoreID = source.StoreID,
        target.StoreName = source.StoreName,
        target.Quantity = source.Quantity,
        target.UnitPrice = source.UnitPrice,
        target.DiscountPct = source.DiscountPct,
        target.GrossSalesAmount = source.GrossSalesAmount,
        target.DiscountAmount = source.DiscountAmount,
        target.NetSalesAmount = source.NetSalesAmount,
        target.SourceSystem = source.SourceSystem,
        target.BatchID = source.BatchID,
        target.LoadDate = source.LoadDate
WHEN NOT MATCHED THEN
    INSERT (
        OrderID,
        OrderLineID,
        OrderDate,
        CustomerID,
        CustomerName,
        ProductID,
        ProductName,
        CategoryName,
        StoreID,
        StoreName,
        Quantity,
        UnitPrice,
        DiscountPct,
        GrossSalesAmount,
        DiscountAmount,
        NetSalesAmount,
        SourceSystem,
        BatchID,
        LoadDate
    )
    VALUES (
        source.OrderID,
        source.OrderLineID,
        source.OrderDate,
        source.CustomerID,
        source.CustomerName,
        source.ProductID,
        source.ProductName,
        source.CategoryName,
        source.StoreID,
        source.StoreName,
        source.Quantity,
        source.UnitPrice,
        source.DiscountPct,
        source.GrossSalesAmount,
        source.DiscountAmount,
        source.NetSalesAmount,
        source.SourceSystem,
        source.BatchID,
        source.LoadDate
    );

-- Check target result after MERGE
SELECT *
FROM dbo.tgt_RetailSales
ORDER BY OrderID, OrderLineID;

STEP 9: CREATE A WAREHOUSE-FRIENDLY MATERIALIZED VIEW PATTERN

Creates a Fabric Warehouse-friendly materialized view pattern using a persisted aggregate table plus a standard view.

-- ============================================================
-- STEP 9: CREATE A WAREHOUSE-FRIENDLY MATERIALIZED VIEW PATTERN
-- ============================================================
-- Important Fabric note:
--   Fabric Warehouse does not use CREATE MATERIALIZED VIEW in this lab.
--   Instead, this lab materializes an aggregation into a physical table.
--   A standard view is then created on top of that table for easy access.
--
-- Pattern:
--   Target table -> Persisted aggregate table -> Standard view

CREATE TABLE dbo.mv_DailySalesSummary (
    OrderDate        DATE          NOT NULL,
    CategoryName     VARCHAR(100)  NOT NULL,
    StoreName        VARCHAR(200)  NOT NULL,
    TotalOrders      INT           NOT NULL,
    TotalUnits       INT           NOT NULL,
    GrossRevenue     DECIMAL(10,2) NOT NULL,
    TotalDiscount    DECIMAL(10,2) NOT NULL,
    NetRevenue       DECIMAL(10,2) NOT NULL
);

INSERT INTO dbo.mv_DailySalesSummary
SELECT
    OrderDate,
    CategoryName,
    StoreName,
    COUNT(DISTINCT OrderID) AS TotalOrders,
    SUM(Quantity) AS TotalUnits,
    SUM(GrossSalesAmount) AS GrossRevenue,
    SUM(DiscountAmount) AS TotalDiscount,
    SUM(NetSalesAmount) AS NetRevenue
FROM dbo.tgt_RetailSales
GROUP BY OrderDate, CategoryName, StoreName;

CREATE VIEW dbo.vw_DailySalesSummary AS
SELECT
    OrderDate,
    CategoryName,
    StoreName,
    TotalOrders,
    TotalUnits,
    GrossRevenue,
    TotalDiscount,
    NetRevenue
FROM dbo.mv_DailySalesSummary;

-- Query the performance-oriented access layer
SELECT *
FROM dbo.vw_DailySalesSummary
ORDER BY OrderDate, CategoryName, StoreName;

STEP 10: REFRESH THE PERSISTED AGGREGATE TABLE

Refreshes the persisted aggregate table after target data changes.

-- ============================================================
-- STEP 10: REFRESH THE PERSISTED AGGREGATE TABLE
-- ============================================================
-- Refresh definition:
--   Because the aggregate table is physically stored, it must be refreshed
--   when the curated target changes.
--
-- Simple classroom refresh pattern:
--   1) Delete old aggregate rows
--   2) Reinsert current aggregate output from the target table

DELETE FROM dbo.mv_DailySalesSummary;

INSERT INTO dbo.mv_DailySalesSummary
SELECT
    OrderDate,
    CategoryName,
    StoreName,
    COUNT(DISTINCT OrderID) AS TotalOrders,
    SUM(Quantity) AS TotalUnits,
    SUM(GrossSalesAmount) AS GrossRevenue,
    SUM(DiscountAmount) AS TotalDiscount,
    SUM(NetSalesAmount) AS NetRevenue
FROM dbo.tgt_RetailSales
GROUP BY OrderDate, CategoryName, StoreName;

SELECT *
FROM dbo.vw_DailySalesSummary
ORDER BY OrderDate, CategoryName, StoreName;
